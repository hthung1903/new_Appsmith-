SELECT
    EXTRACT(YEAR FROM b.NgayBanHanh) AS Nam,
    COUNT(DISTINCT CASE WHEN EXTRACT(QUARTER FROM b.NgayBanHanh) = 1 THEN bll.MucTieuID END) AS Q1,
    COUNT(DISTINCT CASE WHEN EXTRACT(QUARTER FROM b.NgayBanHanh) = 2 THEN bll.MucTieuID END) AS Q2,
    COUNT(DISTINCT CASE WHEN EXTRACT(QUARTER FROM b.NgayBanHanh) = 3 THEN bll.MucTieuID END) AS Q3,
    COUNT(DISTINCT CASE WHEN EXTRACT(QUARTER FROM b.NgayBanHanh) = 4 THEN bll.MucTieuID END) AS Q4
FROM
    BaoCao b
JOIN
    BaoCaoLoLot bll ON b.ID = bll.BaoCaoID
WHERE
    b.LoaiBaoCao = 'Báo cáo lộ lọt tài khoản'
    AND b.NgayBanHanh BETWEEN '{{DatePickerStart4_1.selectedDate}}' AND '{{DatePickerEnd1_1.selectedDate}}'
GROUP BY
    EXTRACT(YEAR FROM b.NgayBanHanh)
ORDER BY
    Nam;
